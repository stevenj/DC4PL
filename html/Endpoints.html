def add(data)
    # Get the endpoint ID
    ep = self.get_endpoint(data["method"],data["path"])

    if ep.nil? then
        # Unknown Endpoint, so just count the number of lines we throw
        @unprocessed_lines += 1
        return false
    end

    ep.add(data)
    return true
end

def get_endpoint(method,path)
    @endpoints.each { | ep |
        return ep[1] if ep[1].is_endpoint?(method,path)
    }
    return nil
end
